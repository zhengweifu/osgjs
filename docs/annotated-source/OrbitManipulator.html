            for ( var i = 0, l = this._current.length; i < l; i++ ) {

        update: function () {
            for ( var i = 0, l = this._current.length; i < l; i++ ) {
                var d = ( this._target[ i ] - this._current[ i ] ) * this._delay;
                this._delta[ i ] = d;
                this._current[ i ] += d;

        },
        set: function () {
